# === ROUTER 1 CONFIGURATION ===
# This connects the server and routes traffic back to the PC VLANs
enable
configure terminal
interface fa0/0
 ip address 10.10.10.1 255.255.255.0
 no shutdown
exit
# Static routes to reach the PC VLAN networks via Router0
ip route 192.168.10.0 255.255.255.0 <Router0_Serial_IP>
ip route 192.168.20.0 255.255.255.0 <Router0_Serial_IP>
ip route 192.168.30.0 255.255.255.0 <Router0_Serial_IP>
exit










# === CONFIGURATION FOR ROUTER 2 ===
enable
configure terminal

# 1. Create Extended ACL 101
# Permit returning TCP traffic for sessions already started by the Secure Network
access-list 101 permit tcp any 192.168.100.0 0.0.0.255 established

# Deny any new TCP connection attempts initiated by internal VLANs or the Server
access-list 101 deny tcp any 192.168.100.0 0.0.0.255

# Permit all other necessary traffic (ICMP/Pings, UDP, etc.)
access-list 101 permit ip any any

# 2. Apply the ACL to the Serial interface facing the core network (Router 0)
# This filters traffic as it enters Router 2 from the outside
interface se2/0
 ip access-group 101 in
exit

# 3. Ensure Routing is active so the Secure Network can find the others
# (Replace se2/0 with the correct exit interface if different in your lab)
ip route 192.168.10.0 255.255.255.0 se2/0
ip route 192.168.20.0 255.255.255.0 se2/0
ip route 192.168.30.0 255.255.255.0 se2/0
ip route 10.10.10.0 255.255.255.0 se2/0





# Map Server 0 (10.10.10.201) to Public IP 20.20.20.200
Router1(config)# ip nat inside source static 10.10.10.201 20.20.20.200
